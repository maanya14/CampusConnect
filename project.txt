create table passenger(
pid int primary key,
pname varchar(30));

create table cab_booking(
booking_id int primary key,
p_id int,
start_city varchar(10),
end_city varchar(10),
booking_date date,
foreign key(p_id) references passenger(pid));

create table Good_BC_City(
good_count int,
good_city varchar(10));


create table Poor_BC_City(
poor_count int,
poor_city varchar(10));

create table post(
post_id int auto_increment primary key,
content text not null,
likes int default 0);
